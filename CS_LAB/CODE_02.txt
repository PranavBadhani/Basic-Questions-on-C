// 02: Write a program in C to find the largest and the second largest element in an array.

// Code:

#include <stdio.h>

int main() {
    int a[10],j;
    printf("Input array elements:\n");
    for(int i=0;i<10;i++)
    {
        scanf("%d",&a[i]);
    }
    int max=a[0];
    printf("\nLargest number is:");
    for(int i=0;i<10;i++)
      {
        if(a[i]>max)
        {
          max=a[i];
          j=i;
        }
        else
        {
          continue;
        }
      }
    printf("\n%d",max);
    int max2=a[0];
    printf("\nSecond largest number is:");
    for(int i=0;i<10;i++)
    {
      if(i==j)
        {
         continue; 
        }
      else
      {
        if (a[i]>max2)
        {
        max2=a[i];
        }
      }
    }
    printf("\n%d",max2);
    return 0;
}

// Output

Input array elements:
1
22
3
44
5

77
8
99
11

Largest number is:
99

Second largest number is:
77
