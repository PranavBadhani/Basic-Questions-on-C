// 9: Write a program in C enter an element at a particular position in an array.

// Code

#include<stdio.h>
int main()
{
    int n,p;
    int a[10],i;
    printf("Enter array elements:\n");
    for(i=0;i<10;i++)
    {
        scanf("%d",&a[i]);

    }
    printf("Elements of an array are:\n");
    for(i=0;i<10;i++)
    {
        printf("%d ",a[i]);
    }
    printf("\nEnter the element \n");
    scanf("%d",&n);
    printf("Enter the position \n");
    scanf("%d",&p);
    for(i=9;i>=p-1;i--)
    {
        a[i+1]=a[i];
    }
    a[p-1]=n;
    printf("The elements of an array are:\n");
    for(i=0;i<10;i++)
    {
        printf("%d ",a[i]);
    }
  return 0;
}
