// 03: Write a program in C to find the smallest and the second smallest element in an array.

// Code

#include <stdio.h>

int main() {
    int a[10],j;
    printf("Input array elements:\n");
    for(int i=0;i<10;i++)
    {
        scanf("%d",&a[i]);
    }
    int min=a[0];
    printf("\nSmallest number is:");
    for(int i=0;i<10;i++)
      {
        if(a[i]<min)
        {
          min=a[i];
          j=i;
        }
        else
        {
          continue;
        }
      }
    printf("\n%d",min);
    int min2=a[0];
    printf("\nSecond smallest number is:");
    for(int i=0;i<10;i++)
    {
      if(i==j)
        {
         continue; 
        }
      else
      {
        if (a[i]<min2)
        {
        min2=a[i];
        }
      }
    }
    printf("\n%d",min2);
    return 0;
}

// Output

Input array elements:
1
2
3
4

9
0

0

Second smallest number is:
1
