// 18: Write a program in C to print product of all elements using function.

// Code:

#include<stdio.h>
int product(int );
int main()
{
    int a[3][3],m;
    int multi=1;
    printf("Enter the elements\n");
    for(int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            scanf("%d",&a[i][j]);
            m=product(a[i][j]);
            multi=multi*m;
        }
    }
    printf("\n%d",multi);
}
int product(int n){
   int  mul=1;
    mul=mul*n;
    return mul;
}
